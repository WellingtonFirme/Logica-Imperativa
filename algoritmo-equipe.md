# Estrutura de repetição



### Exemplo "fim enquanto"

var 

contador, numero: inteiro



inicio

contador <- 1

enquanto(contador<=10)faca

escreva("digite um número:")

leia(numero)

escreva("o número digitado foi:", numero)

contador <- contador +1

finenquanto



### Exempo "repita até"

var

numero,contador: inteiro

inicio

contador <-1

repita

escreva("digite um número:")

leia(numero)

escreva("o número digitado foi : ", numero)

ate (contador >10)
